    def share_directory():

        path = os.path.dirname(Path.CodeReview_module_directory)
        if path.startswith(sys.exec_prefix):
            return os.path.join(sys.exec_prefix, 'share', 'CodeReview')
        else:
            return os.path.join(path, 'share')
